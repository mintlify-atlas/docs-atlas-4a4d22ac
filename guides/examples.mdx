## Checkout Example

Create a complete checkout flow with items, buyer information, and redirect URLs.

```javascript
import React, { Component } from 'react';
import paymaya from 'paymaya-js-sdk';

class Checkout extends Component {
  constructor(props) {
    super(props);
    this.state = {
      loading: false,
      requestReferenceNumber: "6319921",
      items: [
        {
          name: "Nike Footwear",
          quantity: 1,
          totalAmount: {
            value: 100
          }
        }
      ],
      totalAmount: {
        value: 100,
        currency: 'PHP',
      },
      metadata: {},
      redirectUrl: {
        success: "http://localhost:3000/success",
        failure: "http://localhost:3000/failure",
        cancel: "http://localhost:3000/cancel"
      },
      errorResponse: {}
    };
  }

  async createCheckout() {
    this.setState({
      loading: true,
      errorResponse: {}
    });

    const { requestReferenceNumber, totalAmount, items, metadata, redirectUrl } = this.state;

    const checkoutData = {
      requestReferenceNumber,
      totalAmount,
      items,
      metadata,
      redirectUrl
    };

    paymaya.init('pk-Z0OSzLvIcOI2UIvDhdTGVVfRSSeiGStnceqwUE7n0Ah', true);
    
    await paymaya.createCheckout(checkoutData)
      .then(() => {
        // User will be redirected to PayMaya checkout
      })
      .catch((err) => {
        this.setState({
          errorResponse: err,
          loading: false
        });
      });
  }

  render() {
    const { loading } = this.state;
    
    return (
      <div className="form">
        <h2>Checkout</h2>
        <button 
          onClick={() => this.createCheckout()} 
          type="button"
          disabled={loading}
        >
          {loading ? 'Processing...' : 'Proceed to Checkout'}
        </button>
      </div>
    );
  }
}

export default Checkout;
```

## Credit Card Form Example

Embed a secure credit card form and handle tokenization.

```javascript
import React, { Component } from 'react';
import paymaya from 'paymaya-js-sdk';

class CreditCard extends Component {
  constructor(props) {
    super(props);
    this.state = {
      paymentTokenId: null
    };
  }

  componentDidMount() {
    this.createCreditCardForm();
  }

  createCreditCardForm = () => {
    paymaya.init('pk-MOfNKu3FmHMVHtjyjG7vhr7vFevRkWxmxYL1Yq6iFk5', true);
    
    const iframeContainer = document.getElementById("iframe-container");
    
    paymaya
      .createCreditCardForm(iframeContainer)
      .addTransactionHandler((paymentTokenId) => {
        console.log('Payment token received:', paymentTokenId);
        this.setState({ paymentTokenId });
        // Use the token to charge the card via your backend
      });
  }

  render() {
    return (
      <div className="form">
        <h2>Credit Card Payment</h2>
        <div id="iframe-container" />
        {this.state.paymentTokenId && (
          <p>Token: {this.state.paymentTokenId}</p>
        )}
      </div>
    );
  }
}

export default CreditCard;
```

```html
          {

    }
  </script>

```

```bash

cd PayMaya-JS-SDK-v2
npm ci
cd example
npm ci
npm start
```
